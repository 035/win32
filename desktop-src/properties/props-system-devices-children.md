# System.Devices.Children

Device instance ids of children of this device.

propertyDescription
   name = System.Devices.Children
   shellPKey = PKEY_Devices_Children
   formatID = 4340A6C5-93FA-4706-972C-7B648008A5A7
   propID = 9
